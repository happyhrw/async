### 冒泡排序

```c
#include "stdio.h"
main()
{
    int a[100],n,i,j,temp;
    scanf("%d",&n);
    for(i=0;i<n;i++)
    scanf("%d",&a[i]);
    for(i=1;i<n;i++)
       for(j=0;j<n-i;j++)
          if(a[j]>a[j+1])
          {
              temp=a[j];
              a[j]=a[j+1];
              a[j+1]=temp;
          }     
        for(i=0;i<n;i++)
            printf("%5d",a[i]);
}
```

### 最大公约数

```c
#include <stdio.h>

int main() {
    int num1, num2, i, gcd;
    printf("请输入两个正整数：\n");
    scanf("%d %d", &num1, &num2);
    for (i = 1; i <= num1 && i <= num2; ++i) {
        if (num1 % i == 0 && num2 % i == 0) {
            gcd = i;
        }
    }
    printf("%d和%d的最大公约数是%d\n", num1, num2, gcd);
    return 0;
}
```

### 菲波那切数列

```c
#include "stdio.h"
int fib(int n)  
{
    int s;
    if(n==1||n==2) s=1;
    else s=fib(n-1)+fib(n-2);
    return s;
}
main()
{	
    int n;
    scanf("%d",&n);
    printf("%d",fib(n));
}
```

### 百钱白鸡

```c
#include "stdio.h"

{
     int x,y,z,c;
     for(x=0,c=0;x<=20;x++) //公鸡不可能超过20只
        for(y=0;y<=33;y++)  //母鸡不可能超过33只
```

### 数组元素颠倒存放

```c
#include "stdio.h"

{						##从小到大输入数组，元素颠倒存放在元素组中						
    int a[10],i,j,t;
    for(i=0;i<10;i++)
        scanf("%d",&a[i]);
    for(i=0,j=9;i<=j;j++,j--)

        printf("%5d",a[i]);
```

### 完数

```c
{						##完数：该数的所有真因子之和等于该数的和
    int i,s,j;			## 如：6=1+2+3    本题求1000以内
```

### 三个数从大到小排序

```c
#include "stdio.h"
main()
{ 	int x,y,z,t;
	scanf("%d%d%d",&x,&y,&z);
	if(x<y){t=x;x=y;y=t;}
	if(x<z){t=x;x=z;z=t;}
	if(y<z){t=z;z=y;y=t;}
	printf("%5d%5d%5d",x,y,z);
}
```

### 玫瑰花数

```C

#include <math.h>

int main() {
    int i, j, k;
    for (i = 1; i <= 9; i++) {
        for (j = 0; j <= 9; j++) {
            for (k = 0; k <= 9; k++) {
```
