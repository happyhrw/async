### 冒泡排序

```c
#include "stdio.h"
main()
{
    int a[100],n,i,j,temp;
    scanf("%d",&n);
    for(i=0;i<n;i++)
    scanf("%d",&a[i]);
    for(i=1;i<n;i++)
       for(j=0;j<n-i;j++)
          if(a[j]>a[j+1])
          {
              temp=a[j];
              a[j]=a[j+1];
              a[j+1]=temp;
          }     
        for(i=0;i<n;i++)
            printf("%5d",a[i]);
}
```

### 菲波那切数列

```c
#include "stdio.h"
int fib(int n)  
{
    int s;
    if(n==1||n==2) s=1;
    else s=fib(n-1)+fib(n-2);
    return s;
}
main()
{	
    int n;
    scanf("%d",&n);
    printf("%d",fib(n));
}
```

{

### 百钱白鸡

```c
#include "stdio.h"
main()
{
     int x,y,z,c;
     for(x=0,c=0;x<=20;x++) //公鸡不可能超过20只
        for(y=0;y<=33;y++)  //母鸡不可能超过33只
           {
               z=100-x-y;
               if(5*x+3+y+z/3==100 && z%3==0) //小鸡1元三只，z必须是3的倍数
               {
                   printf("%d,%d,%d\n",x,y,z);
                   c++;               
               }           
           } 
          if(c==0) printf("无解");//防止没有输出  
}
```

### 数组元素颠倒存放

```c
#include "stdio.h"
main()
{						##从小到大输入数组，元素颠倒存放在元素组中						
    int a[10],i,j,t;
    for(i=0;i<10;i++)
        scanf("%d",&a[i]);
    for(i=0,j=9;i<=j;j++,j--)
    {t=a[i];a[i]=a[j];a[j]=t;}
    for(i=0;i<10;i++)
        printf("%5d",a[i]);
}
```

### 完数

```c
#include "stdio.h"
main()
{						##完数：该数的所有真因子之和等于该数的和
    int i,s,j;			## 如：6=1+2+3    本题求1000以内
    for(i=2;i<=1000;i++)
    {
        s=0;
        for(j=1;j<=i-1;j++)
            if(i%j==0) s+=j;
        	if(i==s) printf("%5d",i);
    }
}
```

### 求1234组成不重复的三位数个数

```c
#include "stdio.h"
main()
{
    int i,j,k,c;
    for(i=1,c=0;i<=4;i++)
        for(j=1;j<=4;j++)
            for(k=1;k<=4;k++)
                if(i!=j&&i!=k&&j!=k)
                    c++;
    printf("%d",c);
}
```
